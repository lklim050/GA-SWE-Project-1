// Create a deck based on card types and their quantities 
// and assign unique IDs to each card
const createDeck = () => {
  deck = [];
  for (let i = 0; i < cardTypes.length; i++) {
    for (let j = 0; j < cardTypesQuantity[i]; j++) {
      let card = { Type: cardTypes[i] };
      deck.push(card);
    }
  }
  for (let i = 0; i < deck.length; i++) {
    deck[i].id = i;
  }
};

// Shuffle the deck using Fisher-Yates algorithm
const shuffleDeck = (deck) => {
  for (let i = deck.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [deck[i], deck[j]] = [deck[j], deck[i]];
  }
};


// Create delay for human like behavior and game flow management
function delay(ms) {
  return new Promise((resolve) => setTimeout(resolve, ms));

// Function to update the dialogue timer visually
const startVisualTimer = (seconds) => {
  let timeLeft = seconds;
  const timerElement = document.getElementById("timer-count");

  const interval = setInterval(() => {
    timeLeft--;
    if (timerElement) timerElement.innerText = timeLeft;
    
    if (timeLeft <= 0) {
      clearInterval(interval);
    }
  }, 1000);

  return interval; // Return so we can stop it if the player clicks early
};

waitForClick in script.js:443 is a generic card picker. It listens on a container, uses event bubbling plus event.target.closest(".card"), and resolves with just the clicked card’s id. It does not care which card type was clicked.

waitForReaction in script.js:476 is a specialized enemy-turn dodge response. It first filters the player hand down to only cards that contain DODGED, attaches click handlers to those specific cards, waits up to a timeout, and resolves with an object like { status: "DODGED_PLAYED", cardId } or null if no dodge was played.

That is why dodge during enemy turn should use waitForReaction, not waitForClick, in script.js:596. The enemy-attack branch expects a DODGED-only reaction and needs the timeout behavior. If you used waitForClick there, any card click could count as a response unless you added extra filtering.
